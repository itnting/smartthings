definition(
    name: "ThermostatMonitor",
    namespace: "doIHaveTo",
    author: "doIHaveTo",
    description: "Monitor the thermostat temprature (and setPointChange) and when it drops below heating setpoint turn on a switch.",
    category: "Convenience",
    iconUrl: "https://s3.amazonaws.com/smartapp-icons/Meta/temp_thermo-switch.png",
    iconX2Url: "https://s3.amazonaws.com/smartapp-icons/Meta/temp_thermo-switch@2x.png"
)

preferences {
	section("Monitor the temperature of...") {
		input "temperatureSensor1", "capability.temperatureMeasurement"
	}
	section( "Notifications" ) {
        input("recipients", "contact", title: "Send notifications to") {
            input "sendPushMessage", "enum", title: "Send a push notification?", options: ["Yes", "No"], required: false
            input "phone1", "phone", title: "Send a Text Message?", required: false
        }
    }
	section("Turn on things...") {
		input "conSwitches", "capability.switch", required: false, multiple:true
	}
}

def installed() {
	subscribe(temperatureSensor1, "temperature", temperatureHandler)
    subscribe(temperatureSensor1, "heatingSetpoint",setPointHandler)
}

def updated() {
	unsubscribe()
	subscribe(temperatureSensor1, "temperature", temperatureHandler)
    subscribe(temperatureSensor1, "heatingSetpoint", setPointHandler)
}

def setPointHandler(evt) {
	log.debug "SetPoint Changed: $evt.value"
    doSwitch(evt)
}

def temperatureHandler(evt) {
	log.debug "Temp Changed: $evt.value"
    doSwitch(evt)
}

def doSwitch(evt) {
	def tempSensor = evt.displayName
    def temp = temperatureSensor1.currentValue("temperature")
    def setpoint = temperatureSensor1.currentValue("heatingSetpoint")
    
    if (temp < setpoint) {

        		log.debug "${tempSensor} Heat:On T:${temp} S:${setpoint} Activating ${conSwitches}"
                conSwitches.on()
                send("${tempSensor} Heat:On T:${temp} S:${setpoint}")

        }
    }
    else if (temp >= setpoint) {
    	def currSwitches = conSwitches.currentSwitch
    	def onSwitches = currSwitches.findAll { it == "on" }
        log.debug "${tempSensor} Heat:On T:${temp} S:${setpoint} - ${conSwitches}"
    	        
        if (onSwitches.size() != 0) {
        	log.debug "${tempSensor} Heat:On T:${temp} S:${setpoint} De-activating ${conSwitches}"
            conSwitches.off()
            send("${tempSensor} H:Off T:${temp} S:${setpoint}")
        }
    }
}

private send(msg) {
    if (location.contactBookEnabled) {
        log.debug("sending notifications to: ${recipients?.size()}")
        sendNotificationToContacts(msg, recipients)
    }
    else {
        if (sendPushMessage != "No") {
            log.debug("sending push message")
            sendPush(msg)
        }

        if (phone1) {
            log.debug("sending text message")
            sendSms(phone1, msg)
        }
    }

    log.debug msg
}
